class KNNRegressor():
  def fit(self,X,y):
    self.X=X
    self.y=y

  def predict(self,X,K,epsilon = 1e-5):
    N = len(X)
    y_hat = np.zeros(N)

    for i in range(N):
      dist2 = np.sum((self.X - X[i])**2,axis=1)
      idxt = np.argsort(dist2)[:K]
      gamma_k=np.exp(-dist2[idxt])/np.exp(-dist2[idxt]).sum()
      y_hat[i] = gamma_k.dot(self.y[idxt])

    return y_hat
